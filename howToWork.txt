### User {

    Admin User : {
        email: admin@gmail.com ,
        password: 123456789
    }

    ## register / Add User <POST>
    https://mini-insta.onrender.com/user/register {
        body: {
            "username": ,
            "email":,
            "password": "",
            "cPassword": "",
            "phone": <optional>,
            "gender": <optional>,
            "role": <optional>
        }
        Authorization {
            Barer Token : Login with Admin User :D
        }
    }

    ## login <POST>
    https://mini-insta.onrender.com/user/login {
        body:{
            "email": Email Must Be Confirmed,
            "password": 
        }
    }

    ## Show Users <GET>
    https://mini-insta.onrender.com/user/allUsers{
        query: {
            page , limit
        }
    }

    ## change Phone <PATCH>
    https://mini-insta.onrender.com/user/updatePhone {
        body: {
            "phone"
        }
        Authorization {
            Barer Token : Login with Same User :D
        }
    }

    ## Change Profile picture <PATCH>
    https://mini-insta.onrender.com/user/addProfilePic {
        body:{
            form-data with Key "avatar"
        }
        Authorization {
            Barer Token : Login with Same User :D
        }
    }

    ## Delete User <DELETE>
    https://mini-insta.onrender.com/user/deleteUser/:userId {
        params: {
            userId
        }
        Authorization {
            Barer Token : Login with Admin User :D
        }
    }
